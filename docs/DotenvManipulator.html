let myObj = {
  "first" : "one",
  "second" : "two",
  "third" : "three",
  "fourth" : "four"
}

dotenvM.bulkAdd(myObj)

OR 

const dotenvM = require('dotenv-manipulator')

dotenvM.add(myObj, (err) => {
  if(err) // could returns : ['second', 'fourth'] if they were already set
    // else ....
})
